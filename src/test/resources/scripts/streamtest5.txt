entry main()
{
	csr = csropen(sifopen("build.xml"));
	each (num, line : csiterate(csr))
		println(num + "\t" + line);
	close(csr);
	
	fin = sifopen("build.xml");
	fout = sofopen("build2.xml");
	println(relay(fin, fout));
	close(fin);
	close(fout);
	println(filelen("build2.xml"));
	println(filedelete("build2.xml"));
}
