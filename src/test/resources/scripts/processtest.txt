#!/usr/bin/env rookscript -i

function cmd(line, work, env)
{
	return exec("cmd.exe", ["/c", line], env, work, STDOUT(), STDERR());
}

check entry main()
{
	println(envvars());
	println(envvars("TMP"));
	println(envvars(["JAVA_HOME", "JRE_PATH"]));
	println(properties());
	println(properties("os.name"));
	println(properties(["os.arch", "os.name"]));
	return cmd("dir", ".", envvars(["path"]))->execresult();
}
