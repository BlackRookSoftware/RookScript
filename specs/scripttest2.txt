#!/usr/bin/env rookscript -i

main()
{
	s = "Matt Tropiano";
	stdout(strupper(s)); 
	stdout(strlower(s)); 
	stdout(strtrim(" apple ")); 
	stdout(strchar(s, 7));
	stdout(substr(s, 5, 0));
	
	stdout(strindex(s, "a"));
	stdout(strlastindex(s, "a"));
	stdout(strindex(s, "x"));
	stdout(strlastindex(s, "x"));
	stdout(strsplit(s, "\\s+"));
	
	list = [];
	
	stdout(listadd(list, 5));
	stdout(listadd(list, 6));
	stdout(listadd(list, 7));
	stdout(listadd(list, 5));
	stdout(list);

	stdout(listaddat(list, 1, 0));
	stdout(listaddat(list, 10, length(list)));
	stdout(listaddat(list, 23, 3));
	stdout(list);

	stdout(listremove(list, 6));
	stdout(list);

	stdout(listremoveat(list, 3));
	stdout(list);

	listsort(list);
	stdout(list);
	
	list2 = listnew(list);
	list3 = list;
	stdout(list);
	stdout(list2);
	stdout(list == list);
	stdout(list == list2);
	stdout(list == list3);

	set1 = set([3,4,5,5,2,1]);
	set2 = set([4,5,6,7,8,9]);
	stdout(set1);
	stdout(set2);
	stdout(setunion(set1, set2));
	stdout(setintersect(set1, set2));
	stdout(setxor(set1, set2));
	stdout(setdiff(set1, set2));

	stdout(setunion(1, 2));
	stdout(setunion(1, [1,2]));
	stdout(setunion([1,2], 1));
	stdout(setintersect(1, 2));
	stdout(setintersect(1, [1,2]));
	stdout(setintersect([1,2], 1));
	stdout(setxor(1, 2));
	stdout(setxor(1, [1,2]));
	stdout(setxor([1,2], 1));
	stdout(setdiff(1, 2));
	stdout(setdiff(1, [1,2]));
	stdout(setdiff([1,2], 1));

}
