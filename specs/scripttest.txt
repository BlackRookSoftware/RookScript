#!/usr/bin/env rookscript -i

main()
{
	out(MIN(6,2));
	out(MAX(6,2));
	out(POW(6,2));
	out(ROUND(6.5));
	out(FLOOR(6.5));
	out(CEILING(6.5));
	out(FIX(234.456, 0));
	out(FIX(234.456, 2));
	out(FIX(234.456, -2));
	out(E());
	out(LOG(e()));
	out(LOG10(10));
	out(SQRT(16));
	out(PI());
	out(DEG2RAD(90));
	out(RAD2DEG(PI()*2));
	out(SIN(0));
	out(COS(0));
	out(TAN(0));
	out(ASIN(1));
	out(ACOS(1));
	out(ATAN(1));
	out(CLAMP(7,0,4));
	out(WRAP(7,0,4));
	out(LERP(0.6,0, 10));
	out(TOBOOLEAN(34));
	out(TOINT(34.6));
	out(TOFLOAT(34));
	out(TOSTRING(34));
	out(PARSEINT(34,5));
	out(COLOR(255,255,255,255));
	out(COLORF(1.0,1.0,1.0,1.0));
}
