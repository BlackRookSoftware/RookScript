function stuff(a)
{
	return a;
} 

function test(a,b,c)
{
	stdout("a:"+a);
	stdout("b:"+b);
	stdout("c:"+c);
} 

function morestuff()
{
}

entry main()
{
	stdout(stuff(5));
	stdout(stuff());
	morestuff()->stdout();
	test(1,2,3);
	test(1,2);
	test(1);
	test();
}
