#!/usr/bin/env rookscript -i

function stuff(a)
{
	return a;
} 

function test(a,b,c)
{
	print("a:"+a);
	print("b:"+b);
	print("c:"+c);
} 

function morestuff()
{
}

entry main()
{
	print(stuff(5));
	print(stuff());
	morestuff()->print();
	test(1,2,3);
	test(1,2);
	test(1);
	test();
}
